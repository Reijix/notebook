## Konjugation
&z = a + bj&
&overlinez = a - bj&
## Absolutbetrag
&abs(z) = sqrt(a^2 + b^2)&
&abs(z)^2 = z * overlinez&
## Polarkoordinaten
&z = (a, b) = (r cos(varphi), r sin(varphi)) = re^(jvarphi)&
&tan(varphi) = b/a&
&cos(varphi) = a/r&
&sin(varphi) = b/r&
## Exponentialfunktion
&e^z = e^(x+jy) = e^x d^(jy)&
## Sinus
&sinx = (e^(jx) - e^(-jx))/(2j)&
&cosz = (e^(jz) + e^(-jz))/(2j)&
